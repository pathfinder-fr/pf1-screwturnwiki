Cette page vous permet de choisir les caractéristiques de votre personnage en utilisant la méthode d'achat de points (version Pathfinder).

===Étape 1===
Choisissez tout d'abord la race de votre personnage. S'il s'agit d'un humain, d'un demi-elfe ou d'un demi-orque, il vous faudra également choisir la caractéristique qui recevra un bonus de +2. Les personnages des autres races, eux, ont des bonus (et un malus) fixé.

===Étape 2===

Ajustez les valeurs des caractéristiques en gardant un oeil sur le coût en points indiqué à droite. Pour modifier les caractéristiques, utilisez les boutons "plus" et "moins" situés à droite. Vous pouvez encore modifier la race ci-dessus si vous le désirez, ou encore changer la caractéristique qui reçoit un bonus de +2 en la choisissant ci-dessus ou en cliquant directement dans le tableau. Vous pouvez également cliquer sur un des boutons "PNJ simple" ou "PNJ héroïque" pour avoir un point de départ ou utiliser une distribution standard.

var val = new Array();
var mod = new Array();
var cout = 0;
initprog();
raz();

function setAll(a,v1,v2,v3,v4,v5,v6) {
  arrayset(a,1,v1);
  arrayset(a,2,v2);
  arrayset(a,3,v3);
  arrayset(a,4,v4);
  arrayset(a,5,v5);
  arrayset(a,6,v6);
}

function getModForm(valmod) {
  if (valmod >= 0) {
    return ("+" + valmod);
  } else {
    return ("" + valmod);
  }
}

function getMod(valstat) {
  var valmod = Math.floor(valstat/2)-5;
  return (getModForm(valmod));
}

function changeRace(val) {
  var r = arrayget(races,parseInt(val));
  if (r.choix == 1) {
    document.getElementById("bonus").disabled = false;
    document.getElementById("bonus").selectedIndex = 0;
    setAll(mod, 2, 0, 0, 0, 0, 0);
  }
  else {
    document.getElementById("bonus").disabled = true;
    setAll(mod, r.mod1, r.mod2, r.mod3, r.mod4, r.mod5, r.mod6);
  }
  contenu = "";
  if (r.lien != "")
  {
    document.getElementById("bwiki").href = r.lien;
	document.getElementById("bwiki").innerHTML = "Lien vers le wiki";
  }
  else
	document.getElementById("bwiki").innerHTML = "";
  document.getElementById("descRace").innerHTML = r.desc;
  updateVals();
}

function changeMods(val) {
  setAll(mod,0,0,0,0,0,0);
  arrayset(mod,parseInt(val),2);
  updateVals();
}

function updateVals() {
  var i = 1;
  var statval = 0;
  var modcout = 0;
  for (var i = 1; i <= 6; i++) {
    document.getElementById("Stat"+i+"Val").innerHTML = "" + arrayget(val,i);
    document.getElementById("Stat"+i+"Rac").innerHTML = getModForm(arrayget(mod,i));
    statval = arrayget(val,i) + arrayget(mod,i);
    document.getElementById("Stat"+i).innerHTML = "" + statval + " (" + getMod(statval) + ")";
    modcout = coutplus(arrayget(val,i));
    if (modcout > 0) {
      document.getElementById("CommP"+i).innerHTML = "coût pr augmenter : " + modcout;
    } else {
      document.getElementById("CommP"+i).innerHTML = "valeur maximale";
    }
    modcout = gainmoins(arrayget(val,i));
    if (modcout > 0) {
      document.getElementById("CommM"+i).innerHTML = "gain en diminuant : " + modcout;
    } else {
      document.getElementById("CommM"+i).innerHTML = "valeur minimale";
    }
  }
  document.getElementById("cout").innerHTML = "" + cout;
}

function raz() {
  setAll(val,10,10,10,10,10,10);
  setAll(mod,2,0,0,0,0,0);
  cout = 0;
  document.getElementById("race").selectedIndex = 0;
  document.getElementById("bonus").selectedIndex = 0;
  changeRace(0);
}

function coutStat(statval) {
  switch (statval) {
    case 7:
      return (-4);
      break;
    case 8:
    case 9:
    case 10:
    case 11:
    case 12:
    case 13:
      return (statval-10);
      break;
    case 14:
      return (5);
      break;
    case 15:
      return (7);
      break;
    case 16:
      return (10);
      break;
    case 17:
      return (13);
      break;
    case 18:
      return (17);
      break;
  }
}

function calcCout() {
  cout = 0;
  for (var i = 1; i <= 6; i++) {
    cout += coutStat(arrayget(val,i));
  }
}

function setVals(v1,v2,v3,v4,v5,v6) {
  setAll(val,v1,v2,v3,v4,v5,v6);
  calcCout();
  updateVals();
}

function coutplus(statval) {
  if (statval == 18) {
    return (-1);
  } else {
    return (coutStat(statval+1) - coutStat(statval));
  }
}

function gainmoins(statval) {
  if (statval == 7) {
    return (-1);
  } else {
    return (coutStat(statval) - coutStat(statval-1));
  }
}

function boutonplus(stat) {
  if (arrayget(val,stat) < 18) {
    cout += coutplus(arrayget(val,stat));
    arrayset(val,stat,arrayget(val,stat)+1);
    updateVals();
  }
}

function boutonmoins(stat) {
  if (arrayget(val,stat) > 7) {
    cout -= gainmoins(arrayget(val,stat));
    arrayset(val,stat,arrayget(val,stat)-1);
    updateVals();
  }
}

function clickRace(stat) {
  if (!document.getElementById("bonus").disabled) {
    document.getElementById("bonus").selectedIndex = stat-1;
    changeMods(stat);
  }
}

function Race (nom, lien, desc, mod1, mod2, mod3, mod4, mod5, mod6, choix) {
  this.nom = nom;
  if (lien == "AUCUN")
    this.lien = "";
  else if (lien == "")
    this.lien = "http://www.pathfinder-fr.org/Wiki/Pathfinder-RPG." + nom + ".ashx";
  else
    this.lien = lien;
  if (desc == "")
    this.desc = "Choisissez une race ci-contre.";
  else
    this.desc = desc;
  this.mod1 = mod1;
  this.mod2 = mod2;
  this.mod3 = mod3;
  this.mod4 = mod4;
  this.mod5 = mod5;
  this.mod6 = mod6;
  this.choix = choix;
}
